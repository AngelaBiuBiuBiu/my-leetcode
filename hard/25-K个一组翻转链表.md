# K个一组翻转链表

## 1. 迭代

```js
/**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} head
 * @param {number} k
 * @return {ListNode}
 */
// 比如输入链表为：1 -> 2 -> 3 -> 4-> 5 -> 6 -> null，k = 3
var reverseKGroup = function (head, k) {
    if (head === null || head.next === null || k === 1) {
        return head
    }
    let dummyHead = new ListNode(-1)
    dummyHead.next = head
    let begin = dummyHead
    let count = 0
    while (head !== null) {
        count++
        if (count % k === 0) {
            // 第一次进入这个分支时：
            // reverseList实际输入的为 -1 -> 1 -> 2 -> 3 -> 4 的链表
            // 参数begin为 -1，head.next为 4
            // 返回后的begin为逆转后的链表的尾结点 3，head为4
            begin = reverseList(begin, head.next)
            head = begin.next
        } else {
            head = head.next
        }
    }
    return dummyHead.next
}cf
var reverseList = function (begin, end) {
    let cur = begin.next
    let next, first = cur, prev = begin
    while (cur !== end) {
        next = cur.next
        cur.next = prev
        prev = cur
        cur = next
    }
    begin.next = prev
    first.next = cur
    return first
}
```

## 2. 递归

```js
var reverseKGroup = function(head, k) {
    if (head === null) {
        return head
    }
    let start = head, end = head
    for (let i = 0; i < k; i++) {
        if (end === null) {
            return head
        }
        end = end.next
    }
    let newHead = reverse(start, end)
    start.next = reverseKGroup(end, k)
    return newHead
}
// 反转链表
var reverse = function (start, end) {
    let [prev, cur] = [null, start]
    while (cur !== end) {
        [cur.next, prev, cur] = [prev, cur, cur.next]
    }
    return prev
}
```

```js
var reverseKGroup = function (head, k) {
    let cur = head
    let count = 0
    while (cur !== null && count !== k) {
        cur = cur.next
        count++
    }
    if (count === k) {
        cur = reverseKGroup(cur, k)
        while (count-- > 0) {
            let temp = head.next
            head.next = cur
            cur = head
            head = temp
        }
        head = cur
    }
    return head
}
```
